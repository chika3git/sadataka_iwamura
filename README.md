# ディレクトリ構造
- 0_task/                # タスク・やりたいこと
- 1_config/              # 設定・秘密（.env.local など）
- 2_input/               # 入力（受領日ベースで整理）
- 3_docs/                # 仕様・調査・長文
- 4_code/                # 実行コード
- 5_output/              # 成果物・中間・ログ
  - deliverables/        # 共有・納品（必要に応じて作成）
  - derived/             # 中間（定期的に掃除）
  - logs/                # 実行ログ
    - screenshots/       # スクショ
  - archive/             # 過去成果
